- debounce
```js
function debounce (fn, time) => {
    let timer = null
    return function (args) {
        const that = this
        if (timer) {
            clearTimeout(timer)
        }
        timer = setTimeout(() => {
            fn.apply(that, args)
            timer = null
        }, time)
    }
}
```
- throttle
```js
function throttle(fn, time) {
    let timer = null
    return function (args) {
        if (timer) { return }
        const that = this
        fn.apply(that, args)
        timer = setTimeout(() => {
            timer = null
        }, time);
    }
}
```
